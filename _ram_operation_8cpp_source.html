199 /** get indexable element */
200 std::unique_ptr<RamValue> RamAggregate::getIndexElement(RamCondition* c, size_t& element, size_t level) {
201  if (auto* binRelOp = dynamic_cast<RamBinaryRelation*>(c)) {
202  if (binRelOp->getOperator() == BinaryConstraintOp::EQ) {
203  if (auto* lhs = dynamic_cast<RamElementAccess*>(binRelOp->getLHS())) {
204  RamValue* rhs = binRelOp->getRHS();
205  if (lhs->getLevel() == level && (isConstant(rhs) || getLevel(rhs) < level)) {
206  element = lhs->getElement();
207  return binRelOp->takeRHS();
208  }
209  }
210  if (auto* rhs = dynamic_cast<RamElementAccess*>(binRelOp->getRHS())) {
211  RamValue* lhs = binRelOp->getLHS();
212  if (rhs->getLevel() == level && (isConstant(lhs) || getLevel(lhs) < level)) {
213  element = rhs->getElement();
214  return binRelOp->takeLHS();
215  }
216  }
217  }
218  }
219  return std::unique_ptr<RamValue>(nullptr);
220 }
