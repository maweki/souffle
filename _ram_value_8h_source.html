50 class RamIntrinsicOperator : public RamValue {
51 private:
52  /** Operation symbol */
53  FunctorOp operation;
54 
55  /** Arguments of the function */
56  std::vector<std::unique_ptr<RamValue>> arguments;
57 
58 public:
59  template <typename... Args>
60  RamIntrinsicOperator(FunctorOp op, Args... args) : RamValue(RN_IntrinsicOperator), operation(op) {
61  std::unique_ptr<RamValue> tmp[] = {std::move(args)...};
62  for (auto& cur : tmp) {
63  arguments.push_back(std::move(cur));
64  }
65  }
66 
67  RamIntrinsicOperator(FunctorOp op, std::vector<std::unique_ptr<RamValue>> args)
68  : RamValue(RN_IntrinsicOperator), operation(op), arguments(std::move(args)) {}
69 
70  /** Print */
71  void print(std::ostream& os) const override {
72  if (isInfixFunctorOp(operation)) {
73  os << "(";
74  os << join(arguments, getSymbolForFunctorOp(operation), print_deref<std::unique_ptr<RamValue>>());
75  os << ")";
76  } else {
77  os << getSymbolForFunctorOp(operation);
78  os << "(";
79  os << join(arguments, ",", print_deref<std::unique_ptr<RamValue>>());
80  os << ")";
81  }
82  }
83 
84  FunctorOp getOperator() const {
85  return operation;
86  }
87 
88  /** Get values */
89  std::vector<RamValue*> getArguments() const {
90  return toPtrVector(arguments);
91  }
92 
93  const RamValue* getArgument(size_t i) const {
94  assert(i >= 0 && i < arguments.size() && "argument index out of bounds");
95  return arguments[i].get();
96  }
97 
98  size_t getArgCount() const {
99  return arguments.size();
100  }
101 
102  /** Obtain list of child nodes */
103  std::vector<const RamNode*> getChildNodes() const override {
104  std::vector<const RamNode*> res;
105  for (const auto& cur : arguments) {
106  res.push_back(cur.get());
107  }
108  return res;
109  }
110 
111  RamIntrinsicOperator* clone() const override {
112  std::vector<std::unique_ptr<RamValue>> argsCopy;
113  for (auto& arg : arguments) {
114  argsCopy.push_back(std::unique_ptr<RamValue>(arg->clone()));
115  }
116  auto res = new RamIntrinsicOperator(operation, std::move(argsCopy));
117  return res;
118  }
119 
120  /** Apply mapper */
121  void apply(const RamNodeMapper& map) override {
122  for (auto& arg : arguments) {
123  arg = map(std::move(arg));
124  }
125  }
126 
127 protected:
128  /** Check equality */
129  bool equal(const RamNode& node) const override {
130  assert(nullptr != dynamic_cast<const RamIntrinsicOperator*>(&node));
131  const auto& other = static_cast<const RamIntrinsicOperator&>(node);
132  return getOperator() == other.getOperator() && equal_targets(arguments, other.arguments);
133  }
